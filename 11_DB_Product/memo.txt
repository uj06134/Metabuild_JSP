drop sequence seqprd;
create sequence seqprd 
	start with 1
	minvalue 1
	increment by 1
	nocache;
	
drop table products cascade constraints purge ;
create table products(
	id number primary key,
	name varchar2(30),
	price number default 0,
	company varchar2(30),
	regdate date default sysdate
) ;
